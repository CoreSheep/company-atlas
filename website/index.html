Example Usage

Python
import requests

# Search companies by industry
response = requests.get(
    "http://localhost:8000/api/v1/companies",
    params={
        "industry": "Technology",
        "country": "USA",
        "min_employees": 100,
        "page": 1,
        "page_size": 10
    }
)

companies = response.json()
print(f"Found {companies['total']} companies")
for company in companies['companies']:
    print(f"- {company['company_name']} ({company['domain']})")
cURL
# Get statistics
curl "http://localhost:8000/api/v1/statistics"

# Search companies
curl "http://localhost:8000/api/v1/companies?industry=Technology&country=USA"

# Get specific company
curl "http://localhost:8000/api/v1/companies/{company_id}"
